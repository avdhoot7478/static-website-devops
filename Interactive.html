HTML
<!DOCTYPE html>
<html>
<head>
    <title>Interactive HTML Page</title>
    <style>
        body {
            font-family: Arial, sans-serif;
            text-align: center;
            padding: 50px;
            transition: background-color 0.5s;
        }

        .container {
            background: white;
            padding: 20px;
            border-radius: 10px;
            width: 350px;
            margin: auto;
            box-shadow: 0 0 10px rgba(0,0,0,0.2);
        }

        input {
            padding: 10px;
            width: 80%;
            margin: 10px 0;
        }

        button {
            padding: 10px 15px;
            margin: 5px;
            cursor: pointer;
        }

        #message {
            margin-top: 15px;
            font-size: 18px;
            font-weight: bold;
            color: #333;
        }
    </style>
</head>
<body>

    <div class="container">
        <h2>Interactive HTML Page</h2>

        <input type="text" id="name" placeholder="Enter your name">

        <br>

        <button onclick="sayHello()">Say Hello</button>
        <button onclick="changeColor()">Change Background</button>

        <div id="message"></div>
    </div>

    <script>
        function sayHello() {
            let name = document.getElementById("name").value;

            if (name === "") {
                document.getElementById("message").innerHTML =
                    "Please enter your name!";
            } else {
                document.getElementById("message").innerHTML =
                    "Hello, " + name + "! Welcome 🎉";
            }
        }

        function changeColor() {
            const colors = [
                "#FFCDD2",
                "#C8E6C9",
                "#BBDEFB",
                "#FFF9C4",
                "#D1C4E9",
                "#B2DFDB"
            ];

            let randomColor =
                colors[Math.floor(Math.random() * colors.length)];

            document.body.style.backgroundColor = randomColor;
        }
    </script>

</body>
</html>

Features

  • Takes user input (name)

  • Displays a personalized greeting

  • Changes the page background color randomly

  • Uses HTML, CSS, and JavaScript in a single file

  • Beginner-friendly and easy to customize
